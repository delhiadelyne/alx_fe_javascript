// --- Posting Data to a Mock API (Optional Enhancement) --- async function postQuoteToServer(quote) { try { const response = await fetch("https://jsonplaceholder.typicode.com/posts", { method: "POST", headers: { "Content-Type": "application/json", }, body: JSON.stringify(quote), }); const result = await response.json(); console.log("Quote successfully posted to server:", result); } catch (error) { console.error("Error posting quote to server:", error); } }
